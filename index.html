    async function saveSignup(type, value) {

      const signups = JSON.parse(localStorage.getItem('halal_signups') || '[]');

      if (GOOGLE_SCRIPT_URL) {

            method: 'POST',

            mode: 'no-cors',

            headers: { 'Content-Type': 'application/json' },

            body: JSON.stringify({ type, value, date: new Date().toISOString() })

        }

      }

    async function submitHeroForm() {

      const input = document.getElementById('heroEmail');

      const val = input.value;

      const isEmail = /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val);

      const isPhone = /[\d\-+()\s]{7,}/.test(val);

      if (!val.trim() || (!isEmail && !isPhone)) {

        input.style.borderColor = '#EF4444';

        input.focus();

        setTimeout(() => input.style.borderColor = '', 1500);

        return;

      }

      const mode = isEmail ? 'email' : 'phone';

      await saveSignup(mode, val);

      document.getElementById('heroFormRow').style.display = 'none';

      document.querySelector('.hero-form .form-hint').style.display = 'none';

      document.getElementById('heroSuccess').classList.add('show');

    }


    // Scroll fade-in

    const obs = new IntersectionObserver(entries => {

      entries.forEach(e => { if (e.isIntersecting) { e.target.classList.add('visible'); obs.unobserve(e.target) } });

    }, { threshold: .12 });

    document.querySelectorAll('.fade-in').forEach(el => obs.observe(el));


    // Stat counter (only for elements with data-target)

    const sObs = new IntersectionObserver(entries => {

      entries.forEach(e => {

        if (!e.isIntersecting) return;

        const el = e.target, t = +el.dataset.target, d = 1200, s = t / (d / 16);

        let c = 0;

        const timer = setInterval(() => { c = Math.min(c + s, t); el.textContent = Math.floor(c).toLocaleString(); if (c >= t) clearInterval(timer) }, 16);

        sObs.unobserve(el);

      });

    }, { threshold: .3 });

    document.querySelectorAll('[data-target]').forEach(el => sObs.observe(el));

    function toggleNav() {

      const nav = document.getElementById('navLinks');

      const btn = document.getElementById('hamburger');

      const isOpen = nav.classList.toggle('nav-open');

      btn.classList.toggle('open', isOpen);

    }


    document.addEventListener('click', function (e) {

      const nav = document.getElementById('navLinks');

      const btn = document.getElementById('hamburger');

      if (nav.classList.contains('nav-open') && !nav.contains(e.target) && !btn.contains(e.target)) {

        nav.classList.remove('nav-open');

        btn.classList.remove('open');

      }

    });


    // ── Registration Form ──

    let regMode = 'email';


    function switchRegMode(mode, btn) {

      regMode = mode;

      const input = document.getElementById('regInput');

      document.querySelectorAll('#regToggle button').forEach(b => b.classList.remove('active'));

      btn.classList.add('active');

      if (mode === 'phone') {

        input.type = 'tel';

        input.placeholder = '+82 10-1234-5678';

      } else {

        input.type = 'email';

        input.placeholder = 'your@email.com';

      }

      input.value = '';

      input.focus();

    }


    function isValidInput(val, mode) {

      if (!val.trim()) return false;

      if (mode === 'email') return /^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(val);

      return /[\d\-+()\s]{7,}/.test(val);

    }


    async function submitRegForm() {

      const input = document.getElementById('regInput');

      const btn = document.getElementById('regSubmit');

      const val = input.value;

      if (!isValidInput(val, regMode)) {

        input.style.borderColor = '#EF4444';

        input.focus();

        setTimeout(() => input.style.borderColor = '', 1500);

        return;

      }

      btn.disabled = true;

      btn.textContent = 'Submitting...';

      await saveSignup(regMode, val);

      document.getElementById('regFormWrap').style.display = 'none';

      document.querySelector('#register .coupon-badge').style.display = 'none';

      document.getElementById('regSuccess').classList.add('show');

    }
